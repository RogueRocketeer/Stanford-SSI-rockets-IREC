1function [forceout, statedot] = dot(t, state, t_step, T, mass, wind, aerodata, CM)
 2
 3% avoid loading too many things. happens once each round of iterations
 4workspace = load('import2simulink.mat');
 5rocket = workspace.rocket;
 6
 7% the fun part
 8statedot = zeros(6,1);      % initializes an empty vector
 9statedot(1:3) = state(4:6); % vx, vy, omega
10
11% lookup current mass and calculate moment of inertia
12if t <= mass(end, 1)
13    k = 1; tol = t_step;
14    while(1)
15        delta = abs(mass(k,1)-t);
16        if delta < tol; break; end
17        k = k+1;
18    end
19    current_mass = mass(k,2);
20else
21    current_mass = mass(end,2);
22end
23
24I = 0.5*current_mass*rocket.OD.^2/4; % hand waves intensely
25
26% find force/acceleration; r = state(1:3); u = state(4:6)
27[f_x, f_y, moment, drag, lift, CD, CL, aoa, W] = forces(t, t_step, ...
28    state(1:3), state(4:6), T, current_mass, wind, aerodata, rocket, CM);
29forceout = [f_x, f_y, moment, drag, lift, CD, CL, aoa, W];
30
31% find acceleration
32a_x = f_x/current_mass;
33a_y = f_y/current_mass;
34alpha = moment/I;
35statedot(4) = a_x;
36statedot(5) = a_y;
37statedot(6) = alpha;
38
39end
