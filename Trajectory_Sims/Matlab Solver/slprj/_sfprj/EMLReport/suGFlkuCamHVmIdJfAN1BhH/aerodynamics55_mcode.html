1% Aerodynamics
 2% Defines the aerodynamic forces acting on the rocket
 3% Ian Gomez, Rushal Rege 04/11/2017
 4
 5function [Fdrag, Flift] = aerodynamics(r, u, wind, aerodata, rocket)
 6% rev 0 = simple drag model (1/2 rho CD v^2)
 7% rev 1 = use CD table from RASAero
 8% rev 2 = use OpenRocket subsonic drag and call RASAero drag past transonic
 9
10% find angle of attack; see photo for help
11% mathisfun.com/algebra/trig-solving-sas-triangles
12umag = norm(u(1:2));
13phi = 90-r(3);
14resultant = umag.^2 + wind.^2 - 2.*umag.*wind.*cosd(phi);
15lambda = asind(wind.*sind(phi)./resultant);
16aoa = 180 - lambda - phi;
17
18% Calculate atmosphere
19[~, ~, rho, local_c, mu] = getAtmoConditions(r(2));
20
21% Check for mach number
22mach = umag/local_c;
23k = 1; tol = 0.01;
24while(1)
25    delta(k) = abs(mach-aerodata(k,1));
26    if delta(k) < tol; break; end;
27    k = k + 1;
28end
29
30ft2m = 0.3048;
31rocket_length = 11*ft2m;
32Re_rocket_y = rho*umag*rocket_length/mu;
33
34
35% determine what state the rocket is in aerodynamically
36% add rail length to avoid tipping; for now use if faster than rail
37% velocity
38
39rail_length = 20; % m, which is pretty ridiculous
40
41if u(2) >= 0
42    
43    if r(2) <= rail_length
44        aoa = 0;
45    end
46    
47    % Compare angle of attack and use weighted average
48    % aerodata = [RASMa, CD0, CD2, CD4, CL0, CL2, CL4];
49    if abs(aoa) <= 2
50        weight = aoa/2;
51        CD = weight * aerodata(k,3) + (1-weight) * aerodata(k,2);
52        CL = weight * aerodata(k,6) + (1-weight) * aerodata(k,5);
53    elseif abs(aoa) <= 4
54        weight = (aoa-2)/2;
55        CD = weight * aerodata(k,4) + (1-weight) * aerodata(k,3);
56        CL = weight * aerodata(k,7) + (1-weight) * aerodata(k,6);
57    else
58        CD = 1e8; % yikes
59        CL = 1e8; % yikes
60        disp('yo, aoa > 4')
61    end
62    
63    S = pi .* (rocket.OD.^2)./4; % axial surface area
64    
65    % do signs
66    Fdrag = CD .* S .* 0.5 .* rho .* (cosd(aoa) .* umag).^2;
67    Flift = CL .* S .* 0.5 .* rho .* (sind(aoa) .* umag).^2;
68    
69elseif u(2) < rocket.drogue.deploy_u && r(2) > rocket.main.deploy_h
70    Fdrag = 0;
71    Flift = 0;
72else
73    Fdrag = 100;
74    Flift = 0;
75end
76
77end
