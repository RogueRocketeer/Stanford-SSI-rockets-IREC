1%%% Forces
 2
 3% Fdrag, Flift, Thrust curve, and Flight data are provided as inputs
 4% Flight Data consists of rocket dimensions, time, position, vel,
 5% acceleration, angle of attack
 6% Assumption is time starts at t = 0;
 7% u >= 0, haven't hit apogee yet
 8% Earth Centered Earth Fixed coordinate system
 9% rocket contains the mass curve
10
11% Split each force into x and y for earth centered components
12
13function [f_x, f_y, moment] = forces(t, t_step, r, u, T, current_mass, wind, aerodata)
14
15ind = t/t_step;
16
17theta = r(3); % wrt to the vertical (normal to the earth's surface)
18
19% thrust
20% thrust curve assumption -- 2 column tabular input
21% find index for this
22if t < T(end,1)
23    thrust = T(ind,2);
24    Tx = sind(theta)*thrust;
25    Ty = cosd(theta)*thrust;
26else
27    Tx = 0;
28    Ty = 0;
29end
30
31% where do we assign direction / theta and signs
32[Fdrag, Flift] = aerodynamics(r, u, wind, aerodata);
33
34% lift
35Lx = cosd(theta)*Flift;
36Ly = sind(theta)*Flift;
37
38% drag
39Dx = sind(theta)*Fdrag;
40Dy = cosd(theta)*Fdrag;
41
42% gravity
43% mass assumption -- 2 column tabular input
44% mdot proportional to thrust (relate to impulse)
45G = 3.986E14; r_earth = 6378000;
46gravity = (G)/(r_earth + norm(r(1:2))) * current_mass;
47
48% Requires CM and CP distance from bottom of the rocket
49if u >= 0
50    f_x = Tx + Lx - Dx;
51    f_y = Ty + Ly - Dy  - gravity;
52    moment = (Lx - Dx)*cosd(theta)*(CP-CM) + (Dy + Ly)*sind(theta)*(CP-CM);
53else
54    f_x = Dx;
55    f_y = Dy - gravity;
56    moment = 0; %not worth it
57end
58
59end
