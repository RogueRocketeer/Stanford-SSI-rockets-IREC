1% Aerodynamics
 2% Defines the aerodynamic forces acting on the rocket
 3% Ian Gomez, Rushal Rege 04/11/2017
 4
 5function [Fdrag, Flift] = aerodynamics(r, u, aerodata)
 6% rev 0 = simple drag model (1/2 rho CD v^2)
 7% rev 1 = use CD table from RASAero
 8% rev 2 = use OpenRocket subsonic drag and call RASAero drag past transonic
 9
10% find angle of attack; see photo for help
11% mathisfun.com/algebra/trig-solving-sas-triangles
12umag = norm(u(1:2));
13phi = 90-r(3);
14resultant = umag.^2 + wind.^2 - 2.*umag.*wind.*cosd(phi);
15lambda = asind(wind.*sind(phi)./resultant);
16aoa = 180 - lambda - phi;
17
18% Calculate atmosphere
19[~,~,rho,local_c] = getAtmoConditions(r(2));
20
21mach = norm(u(1:2))/local_c;
22
23% Check for mach number;
24k = 1; tol = 0.01;
25while(abs(mach-aerodata(k,1)) > tol)
26    k = k + 1;
27end
28
29% determine what state the rocket is in aerodynamically
30
31if u(2) > 0
32    % Compare angle of attack and use weighted average
33    % aerodata = [RASMa, CD0, CD2, CD4, CL0, CL2, CL4];
34    if abs(aoa) <= 2
35        weight = aoa/2;
36        CD = weight * aerodata(k,3) + (1-weight) * aerodata(k,2);
37        CL = weight * aerodata(k,6) + (1-weight) * aerodata(k,5);
38    elseif abs(aoa) <= 4
39        weight = (aoa-2)/2;
40        CD = weight * aerodata(k,4) + (1-weight) * aerodata(k,3);
41        CL = weight * aerodata(k,7) + (1-weight) * aerodata(k,6);
42    else
43        CD = 1e8; % yikes
44        CL = 1e8; % yikes
45        disp('yo, aoa > 4')
46    end
47    
48    S = pi .* (rocket.OD.^2)./4; % axial surface area
49    Fdrag = CD .* S .* 0.5 .* rho .* (cosd(aoa) .* norm(u(1:2))).^2;
50    Flift = CL .* S .* 0.5 .* rho .* (sind(aoa) .* norm(u(1:2))).^2;
51    
52elseif u(2) < rocket.drogue.deploy_u && r(2) > rocket.main.deploy_h
53    Fdrag = 25;
54    Flift = 0;
55else
56    Fdrag = 100;
57    Flift = 0;
58end
59
60end
