1% Aerodynamics
 2% Defines the aerodynamic forces acting on the rocket
 3% Ian Gomez, Rushal Rege 04/11/2017
 4
 5function [Fdrag, Flift] = aerodynamics(r, u, wind, aerodata, rocket)
 6% rev 0 = simple drag model (1/2 rho CD v^2)
 7% rev 1 = use CD table from RASAero
 8% rev 2 = use OpenRocket subsonic drag and call RASAero drag past transonic
 9
10% find angle of attack; see photo for help
11% mathisfun.com/algebra/trig-solving-sas-triangles
12umag = norm(u(1:2));
13phi = 90-r(3);
14resultant = umag.^2 + wind.^2 - 2.*umag.*wind.*cosd(phi);
15lambda = asind(wind.*sind(phi)./resultant);
16aoa = 180 - lambda - phi;
17
18% Calculate atmosphere
19[~, ~, rho, local_c, mu] = getAtmoConditions(r(2));
20
21% Check for mach number
22mach = umag/local_c;
23k = 1; tol = 0.01;
24while(abs(mach-aerodata(k,1)) > tol)
25    k = k + 1;
26end
27
28ft2m = 0.3048;
29rocket_length = 11*ft2m;
30Re_rocket_y = rho*umag*rocket_length/mu;
31
32
33% determine what state the rocket is in aerodynamically
34% add rail length to avoid tipping; for now use if faster than rail
35% velocity
36
37rail_length = 20; % m, which is pretty ridiculous
38
39if u(2) >= 0
40    
41    if r(2) <= rail_length
42        aoa = 0;
43    end
44    
45    % Compare angle of attack and use weighted average
46    % aerodata = [RASMa, CD0, CD2, CD4, CL0, CL2, CL4];
47    if abs(aoa) <= 2
48        weight = aoa/2;
49        CD = weight * aerodata(k,3) + (1-weight) * aerodata(k,2);
50        CL = weight * aerodata(k,6) + (1-weight) * aerodata(k,5);
51    elseif abs(aoa) <= 4
52        weight = (aoa-2)/2;
53        CD = weight * aerodata(k,4) + (1-weight) * aerodata(k,3);
54        CL = weight * aerodata(k,7) + (1-weight) * aerodata(k,6);
55    else
56        CD = 1e8; % yikes
57        CL = 1e8; % yikes
58        disp('yo, aoa > 4')
59    end
60    
61    S = pi .* (rocket.OD.^2)./4; % axial surface area
62    
63    % do signs
64    Fdrag = CD .* S .* 0.5 .* rho .* (cosd(aoa) .* umag).^2;
65    Flift = CL .* S .* 0.5 .* rho .* (sind(aoa) .* umag).^2;
66    
67elseif u(2) < rocket.drogue.deploy_u && r(2) > rocket.main.deploy_h
68    Fdrag = 0;
69    Flift = 0;
70else
71    Fdrag = 100;
72    Flift = 0;
73end
74
75end
