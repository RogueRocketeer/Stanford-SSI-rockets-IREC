1% getTemperature
 2% Grabs atmospheric temperature at an altitude
 3% www.engineeringtoolbox.com/international-standard-atmosphere-d_985.html
 4% Ian Gomez,  08/18/16
 5
 6% Input = altitude [m]; Output = temperature [K]
 7
 8function [T, P, rho, a, mu] = getAtmoConditions(h)
 9
10T0 = 288.16; % [K] launch site temp
11Ts = [T0, 216.66, 282.66, 165.66, 225.66];
12Trate = [-6.5e-3, 0, 3e-3, 0, -4.5e-3, 0, 4e-3]; % uses standard model lapse rate
13hpos  = [11e3, 25e3, 47e3, 53e3, 79e3, 90e3, 105e3];
14
15if h <= hpos(1)
16    T = Ts(1) + Trate(1)*(hpos(1)-h);
17elseif h <= hpos(2)
18    T = Ts(2) + Trate(2)*(hpos(2)-h);
19elseif h <= hpos(3)
20    T = Ts(3) + Trate(3)*(hpos(3)-h);
21elseif h <= hpos(4)
22    T = Ts(4) + Trate(4)*(hpos(4)-h);
23elseif h <= hpos(5)
24    T = Ts(5) + Trate(5)*(hpos(5)-h);
25elseif h <= hpos(6)
26    T = Ts(6) + Trate(6)*(hpos(6)-h);
27elseif h <= hpos(7)
28    T = Ts(7) + Trate(7)*(hpos(7)-h);
29else
30    T = 5e10; % yikes
31end
32
33% Viscosity - definitely is temperature dependent 
34mu = 1.846e-5; % kg/(m*s)
35
36% Pressure [Pa]
37g = 9.81;                   % m/s^2
38avogadro = 6.0221409e23;
39k_boltz = 1.38064852e-23;   % J/K
40m_air   = 29/avogadro*1e-3; % kg
41R_air =  287;               % J/kg K
42P_sl  = 101300;             % Pa, sea level pressure
43k = 1.4;                    % Air specific heats ratio
44P = P_sl.*exp((-m_air.*g.*h)./(k_boltz.*T)); % Pa
45
46% Density [kg/m^3]
47rho = 1.2250.*(288.15./(288.15-0.0065.*h)).^...
48    (1+9.80665.*0.0289644./(8.31432.*0.0065));
49
50% Mach number
51a  = sqrt(k.*R_air.*T); % m/s, local speed of sound
52
53end
