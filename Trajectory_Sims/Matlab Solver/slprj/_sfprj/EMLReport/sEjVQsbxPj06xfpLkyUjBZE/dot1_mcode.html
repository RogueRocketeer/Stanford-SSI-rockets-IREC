1function statedot = dot(t, t_step, state, rocketOD, T, mass)
 2
 3% if rocket is at apogee, set theta = -180deg
 4
 5ind = t/t_step + 1; % matlab indexes at 1
 6
 7statedot = zeros(6,1);      % initializes an empty vector
 8statedot(1:3) = state(4:6); % vx, vy, omega
 9
10% lookup current mass and calculate moment of inertia
11if t <= mass(end, 1) 
12    current_mass = mass(ind,2);
13else
14    current_mass = mass(end,2);
15end
16
17I = 0.5*current_mass*rocketOD.^2/4; % hand waves intensely
18
19% find force/acceleration; r = state(1:3); u = state(4:6)
20[f_x, f_y, moment] = forces(t, t_step, state(1:3), state(4:6), T,...
21    current_mass);
22
23% find acceleration
24a_x = f_x/current_mass;
25a_y = f_y/current_mass;
26alpha = moment/I;
27statedot(4) = a_x;
28statedot(5) = a_y;
29statedot(6) = alpha;

31end
