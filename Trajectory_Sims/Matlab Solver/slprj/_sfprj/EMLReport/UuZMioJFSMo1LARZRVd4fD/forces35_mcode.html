1%%% Forces
 2
 3% Fdrag, Flift, Thrust curve, and Flight data are provided as inputs
 4% Flight Data consists of rocket dimensions, time, position, vel,
 5% acceleration, angle of attack
 6% Assumption is time starts at t = 0;
 7% u >= 0, haven't hit apogee yet
 8% Earth Centered Earth Fixed coordinate system
 9% rocket contains the mass curve
10
11% Split each force into x and y for earth centered components
12
13function [f_x, f_y, moment, Fdrag, Flift, CD, CL, aoa, gravity, CP] = ...
14    forces(t, t_step, r, u, T, current_mass, wind, aerodata, rocket, CM, ...
15    site_elevation, T0)
16
17ux = u(1);
18theta = r(3); % wrt to the vertical (normal to the earth's surface)
19
20% thrust
21% thrust curve assumption -- 2 column tabular input
22% find index for this
23if t < T(end,1)
24    k = 1; tol = t_step;
25    while(1)
26        delta = abs(T(k,1)-t);
27        if delta < tol; break; end
28        k = k+1;
29    end
30    thrust = T(k,2);
31    Tx = sind(theta)*thrust;
32    Ty = cosd(theta)*thrust;
33else
34    Tx = 0;
35    Ty = 0;
36end
37
38% where do we assign direction / theta and signs
39[Fdrag, Flift, CD, CL, aoa, CP] = aerodynamics(r, u, wind, aerodata, ...
40    rocket, site_elevation, T0);
41
42% lift
43Lx = cosd(theta)*Flift;
44Ly = sind(theta)*Flift;
45
46% drag
47Dx = sind(theta)*Fdrag;
48Dy = cosd(theta)*Fdrag;
49
50% gravity
51% mass assumption -- 2 column tabular input
52% mdot proportional to thrust (relate to impulse)
53G = 3.986E14; r_earth = 6378000;
54gravity = (G)/(r_earth + norm(r(1:2))).^2 * current_mass;
55
56% no gravity at the ground
57if r(2) == site_elevation
58    gravity = 0;
59end
60
61if wind >= ux; dir = 1; else dir = -1; end % change direction of wind
62
63% Requires CM and CP distance from bottom of the rocket
64if u(2) >= 0
65    f_x = Tx + Lx + dir*Dx;
66    f_y = Ty + Ly - Dy  - gravity;
67    moment = (Lx - Dx)*cosd(theta)*(CP-CM) + (Dy + Ly)*sind(theta)*(CP-CM);
68%     moment = (Lx - Dx)*(CP-CM) + (Dy + Ly)*(CP-CM);
69
70else    
71    f_x = Dx + dir*Lx;
72    f_y = Ly + Dy - gravity;
73    moment = 0; % not worth it, rocket would be swinging
74end
75
76% if r(2) < site_elevation
77%     fx = 0; fy = 0; moment = 0;
78% end
79
80end
