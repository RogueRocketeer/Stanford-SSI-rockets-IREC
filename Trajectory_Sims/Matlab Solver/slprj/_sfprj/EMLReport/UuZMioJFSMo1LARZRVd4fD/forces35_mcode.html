1%%% Forces
 2
 3% Fdrag, Flift, Thrust curve, and Flight data are provided as inputs
 4% Flight Data consists of rocket dimensions, time, position, vel,
 5% acceleration, angle of attack
 6% Assumption is time starts at t = 0;
 7% u >= 0, haven't hit apogee yet
 8% Earth Centered Earth Fixed coordinate system
 9% rocket contains the mass curve
10
11% Split each force into x and y for earth centered components
12
13function [f_x, f_y, moment, Fdrag, Flift, CD, CL, aoa, gravity] = forces(t, t_step, r, u, T, current_mass, wind, aerodata, rocket, CP, CM)
14
15theta = r(3); % wrt to the vertical (normal to the earth's surface)
16
17% thrust
18% thrust curve assumption -- 2 column tabular input
19% find index for this
20if t < T(end,1)
21    k = 1; tol = t_step;
22    while(1)
23        delta = abs(T(k,1)-t);
24        if delta < tol; break; end
25        k = k+1;
26    end
27    thrust = T(k,2);
28    Tx = sind(theta)*thrust;
29    Ty = cosd(theta)*thrust;
30else
31    Tx = 0;
32    Ty = 0;
33end
34
35% where do we assign direction / theta and signs
36[Fdrag, Flift, CD, CL, aoa] = aerodynamics(r, u, wind, aerodata, rocket);
37
38% lift
39Lx = cosd(theta)*Flift;
40Ly = sind(theta)*Flift;
41
42% drag
43Dx = sind(theta)*Fdrag;
44Dy = cosd(theta)*Fdrag;
45
46% gravity
47% mass assumption -- 2 column tabular input
48% mdot proportional to thrust (relate to impulse)
49G = 3.986E14; r_earth = 6378000;
50gravity = (G)/(r_earth + norm(r(1:2))).^2 * current_mass;
51
52site_elevation = 1293;
53if r(2) <= site_elevation
54    gravity = 0;
55end
56
57% Requires CM and CP distance from bottom of the rocket
58if u(2) >= 0
59    f_x = Tx + Lx - Dx;
60    f_y = Ty + Ly - Dy  - gravity;
61    moment = (Lx - Dx)*cosd(theta)*(CP-CM) + (Dy + Ly)*sind(theta)*(CP-CM);
62else
63    f_x = Dx;
64    f_y = Dy - gravity;
65    moment = 0; %not worth it
66end
67
68end
