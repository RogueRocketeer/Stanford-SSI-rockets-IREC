1%%% Forces
 2
 3% Fdrag, Flift, Thrust curve, and Flight data are provided as inputs
 4% Flight Data consists of rocket dimensions, time, position, vel,
 5% acceleration, angle of attack
 6% Assumption is time starts at t = 0;
 7% u >= 0, haven't hit apogee yet
 8% Earth Centered Earth Fixed coordinate system
 9% rocket contains the mass curve
10
11% Split each force into x and y for earth centered components
12
13function [f_x, f_y, moment, Fdrag, Flift, CD, CL, aoa, gravity] = ...
14    forces(t, t_step, r, u, T, current_mass, wind, aerodata, rocket, CM)
15
16theta = r(3); % wrt to the vertical (normal to the earth's surface)
17
18% thrust
19% thrust curve assumption -- 2 column tabular input
20% find index for this
21if t < T(end,1)
22    k = 1; tol = t_step;
23    while(1)
24        delta = abs(T(k,1)-t);
25        if delta < tol; break; end
26        k = k+1;
27    end
28    thrust = T(k,2);
29    Tx = sind(theta)*thrust;
30    Ty = cosd(theta)*thrust;
31else
32    Tx = 0;
33    Ty = 0;
34end
35
36% where do we assign direction / theta and signs
37[Fdrag, Flift, CD, CL, aoa, CP] = aerodynamics(r, u, wind, aerodata, rocket);
38
39% lift
40Lx = cosd(theta)*Flift;
41Ly = sind(theta)*Flift;
42
43% drag
44Dx = sind(theta)*Fdrag;
45Dy = cosd(theta)*Fdrag;
46
47% gravity
48% mass assumption -- 2 column tabular input
49% mdot proportional to thrust (relate to impulse)
50G = 3.986E14; r_earth = 6378000;
51gravity = (G)/(r_earth + norm(r(1:2))).^2 * current_mass;
52
53site_elevation = 1293;
54if r(2) <= site_elevation
55    gravity = 0;
56end
57
58if wind >= 0; dir = 1; else dir = -1; end % change direction of wind
59
60% Requires CM and CP distance from bottom of the rocket
61if u(2) >= 0
62    f_x = Tx + Lx + dir*Dx;
63    f_y = Ty + Ly - Dy  - gravity;
64    moment = (Lx - Dx)*cosd(theta)*(CP-CM) + (Dy + Ly)*sind(theta)*(CP-CM);
65else
66    f_x = dir*Dx;
67    f_y = Dy - gravity;
68    moment = 0; %not worth it
69end
70
71end
