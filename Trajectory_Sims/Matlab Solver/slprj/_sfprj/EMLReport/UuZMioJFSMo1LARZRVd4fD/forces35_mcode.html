1%%% Forces
 2
 3% Fdrag, Flift, Thrust curve, and Flight data are provided as inputs
 4% Flight Data consists of rocket dimensions, time, position, vel,
 5% acceleration, angle of attack
 6% Assumption is time starts at t = 0;
 7% u >= 0, haven't hit apogee yet
 8% Earth Centered Earth Fixed coordinate system
 9% rocket contains the mass curve
10
11% Split each force into x and y for earth centered components
12
13function [f_x, f_y, moment, Fdrag, Flift, CD, CL, aoa, gravity] = ...
14    forces(t, t_step, r, u, T, current_mass, wind, aerodata, rocket, CM)
15
16ux = u(1);
17theta = r(3); % wrt to the vertical (normal to the earth's surface)
18
19% thrust
20% thrust curve assumption -- 2 column tabular input
21% find index for this
22if t < T(end,1)
23    k = 1; tol = t_step;
24    while(1)
25        delta = abs(T(k,1)-t);
26        if delta < tol; break; end
27        k = k+1;
28    end
29    thrust = T(k,2);
30    Tx = sind(theta)*thrust;
31    Ty = cosd(theta)*thrust;
32else
33    Tx = 0;
34    Ty = 0;
35end
36
37% where do we assign direction / theta and signs
38[Fdrag, Flift, CD, CL, aoa, CP] = aerodynamics(r, u, wind, aerodata, rocket);
39
40% lift
41Lx = cosd(theta)*Flift;
42Ly = sind(theta)*Flift;
43
44% drag
45Dx = sind(theta)*Fdrag;
46Dy = cosd(theta)*Fdrag;
47
48% gravity
49% mass assumption -- 2 column tabular input
50% mdot proportional to thrust (relate to impulse)
51G = 3.986E14; r_earth = 6378000;
52gravity = (G)/(r_earth + norm(r(1:2))).^2 * current_mass;
53
54% at zero no gravity
55site_elevation = 1293;
56if r(2) == site_elevation
57    gravity = 0;
58end
59
60if wind >= ux; dir = 1; else dir = -1; end % change direction of wind
61
62% Requires CM and CP distance from bottom of the rocket
63if u(2) >= 0
64    f_x = Tx + Lx + dir*Dx;
65    f_y = Ty + Ly - Dy  - gravity;
66    moment = (Lx - Dx)*cosd(theta)*(CP-CM) + (Dy + Ly)*sind(theta)*(CP-CM);
67elseif u(2) < 0    
68    f_x = Dx + dir*Lx;
69    f_y = Ly + Dy - gravity;
70    moment = 0; % not worth it, rocket would be swinging
71end
72
73% if r(2) < site_elevation
74%     fx = 0; fy = 0; moment = 0;
75% end
76
77end
