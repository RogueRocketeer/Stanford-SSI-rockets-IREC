1% Aerodynamics
 2% Defines the aerodynamic forces acting on the rocket
 3% Ian Gomez, Rushal Rege 04/11/2017
 4
 5function [Fdrag, Flift, CD, CL, aoa] = aerodynamics(r, u, wind, aerodata, rocket)
 6% rev 0 = simple drag model (1/2 rho CD v^2)
 7% rev 1 = use CD table from RASAero
 8% rev 2 = use OpenRocket subsonic drag and call RASAero drag past transonic
 9
10% find angle of attack; see photo for help
11% mathisfun.com/algebra/trig-solving-sas-triangles
12umag = norm(u(1:2));
13phi = 90-r(3);
14resultant = umag.^2 + wind.^2 - 2.*umag.*wind.*cosd(phi);
15lambda = asind(wind.*sind(phi)./resultant);
16aoa = 180 - lambda - phi;
17
18% Calculate atmosphere
19[~, ~, rho, local_c, mu] = getAtmoConditions(r(2));
20
21% Check for mach number
22mach = umag/local_c;
23k = 1; tol = 0.01;
24while(1)
25    delta = abs(mach-aerodata(k,1));
26    if delta < tol; break; end;
27    k = k + 1;
28end
29
30ft2m = 0.3048;
31rocket_length = 11*ft2m;
32Re_rocket_y = rho*umag*rocket_length/mu;
33
34% determine what state the rocket is in aerodynamically
35% if rocket is falling - set theta 
36
37rail_length = 20; % m, which is pretty ridiculous
38launch_site = 1293; % m, magic number
39aoa = 0;
40if u(2) >= 0
41    
42    if r(2) <= rail_length+launch_site
43        aoa = 0;
44    end
45    
46    % Compare angle of attack and use weighted average
47    % aerodata = [RASMa, CD0, CD2, CD4, CL0, CL2, CL4];
48    if abs(aoa) <= 2
49        weight = aoa/2;
50        CD = weight * aerodata(k,3) + (1-weight) * aerodata(k,2);
51        CL = weight * aerodata(k,6) + (1-weight) * aerodata(k,5);
52    elseif abs(aoa) <= 4
53        weight = (aoa-2)/2;
54        CD = weight * aerodata(k,4) + (1-weight) * aerodata(k,3);
55        CL = weight * aerodata(k,7) + (1-weight) * aerodata(k,6);
56    else
57        CD = 1e3; % yikes
58        CL = 1e3; % yikes
59        disp('yo, aoa > 4')
60    end
61    
62    S = pi .* (rocket.OD.^2)./4; % axial surface area
63    
64    % do signs
65    Fdrag = CD .* S .* 0.5 .* rho .* (cosd(aoa) .* umag).^2;
66    Flift = CL .* S .* 0.5 .* rho .* (sind(aoa) .* umag).^2;
67    
68elseif u(2) < rocket.drogue.deploy_u && r(2) > rocket.main.deploy_h
69    CD = rocket.drogue.Cd;
70    CL = 0;
71    Fdrag = CD.*rocket.drogue.S.*0.5.*rho.*umag.^2;
72    Flift = CD.*rocket.l.*rocket.OD.*0.5.*rho.*(ux-wind).^2;
73else
74    CD = rocket.main.Cd;
75    CL = 0;
76    Fdrag = CD.*rocket.main.S.*0.5.*rho.*umag.^2;
77    Flift = 0;
78end
79
80end
