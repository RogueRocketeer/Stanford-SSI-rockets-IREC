1% Aerodynamics
 2% Defines the aerodynamic forces acting on the rocket
 3% Ian Gomez, Rushal Rege 04/11/2017
 4
 5function [Fdrag, Flift, CD, CL, aoa, CP] = aerodynamics(r, u, wind, aerodata, rocket)
 6% rev 0 = simple drag model (1/2 rho CD v^2)
 7% rev 1 = use CD table from RASAero
 8% rev 2 = use OpenRocket subsonic drag and call RASAero drag past transonic
 9
10% find angle of attack; see photo for help
11% mathisfun.com/algebra/trig-solving-sas-triangles
12ux = u(1); uy = u(2);
13umag = norm(u(1:2));
14phi = 90-r(3);
15resultant = umag.^2 + wind.^2 - 2.*umag.*wind.*cosd(phi);
16lambda = asind(wind.*sind(phi)./resultant);
17aoa = 180 - lambda - phi;
18
19% Calculate atmosphere
20[~, ~, rho, local_c, mu] = getAtmoConditions(r(2));
21
22% Check for mach number
23mach = umag/local_c;
24k = 1; tol = 0.01;
25while(1)
26    delta = abs(mach-aerodata(k,1));
27    if delta < tol; break; end;
28    k = k + 1;
29end
30
31ft2m = 0.3048;
32rocket_length = 11*ft2m;
33Re_rocket_y = rho*umag*rocket_length/mu;
34
35% determine what state the rocket is in aerodynamically
36% if rocket is falling - set theta
37
38rail_length = 20; % m, which is pretty ridiculous
39launch_site = 1293; % m, magic number
40aoa = 0;
41if u(2) >= 0
42    
43    if r(2) <= rail_length+launch_site
44        aoa = 0;
45    end
46    
47    % Compare angle of attack and use weighted average
48    % aerodata = [RASMa, CD0, CD2, CD4, CL0, CL2, CL4, CP0, CP2, CP4];
49    if abs(aoa) <= 2
50        weight = aoa/2;
51        CD = weight * aerodata(k,3) + (1-weight) * aerodata(k,2);
52        CL = weight * aerodata(k,6) + (1-weight) * aerodata(k,5);
53        CP = weight * aerodata(k,9) + (1-weight) * aerodata(k,8);
54    elseif abs(aoa) <= 4
55        weight = (aoa-2)/2;
56        CD = weight * aerodata(k,04) + (1-weight) * aerodata(k,3);
57        CL = weight * aerodata(k,07) + (1-weight) * aerodata(k,6);
58        CP = weight * aerodata(k,10) + (1-weight) * aerodata(k,9);
59    else
60        CD = 1e10; % yikes
61        CL = 1e10; % yikes
62        CP = 1e10; % yikes
63        disp('yo, aoa > 4')
64    end
65    
66    S = pi .* (rocket.OD.^2)./4; % axial surface area
67    
68    % do signs
69    Fdrag = CD .* S .* 0.5 .* rho .* (cosd(aoa) .* umag).^2;
70    Flift = CL .* S .* 0.5 .* rho .* (sind(aoa) .* umag).^2;
71    
72elseif u(2) < rocket.drogue.deploy_u && r(2) > rocket.main.deploy_h
73    CD = 0;
74    CL = 1;
75    CP = aerodata(1,8);
76    Fdrag = rocket.drogue.Cd.*rocket.drogue.S.*0.5.*rho.*umag.^2;
77    Flift = CL.*rocket.l.*rocket.OD.*0.5.*rho.*(ux-wind).^2; % need to fix
78elseif r(2) < rocket.main.deploy_h
79    CD = 0;
80    CL = 1;
81    CP = aerodata(1,8);
82    Fdrag = rocket.main.Cd.*rocket.main.S.*0.5.*rho.*umag.^2;
83    Flift = CL.*rocket.l.*rocket.OD.*0.5.*rho.*(ux-wind).^2; % need to fix
84else
85    CD = 0;
86    CL = 0;
87    CP = aerodata(1,8);
88    Fdrag = 0;
89    Flift = 0;
90end
91
92end
