1% Aerodynamics
 2% Defines the aerodynamic forces acting on the rocket
 3% Ian Gomez, Rushal Rege 04/11/2017
 4
 5function [Fdrag, Flift, CD, CL, aoa, CP] = aerodynamics(r, u, wind, ...
 6    aerodata, rocket, site_elevation, T0)
 7% rev 2 = use OpenRocket subsonic drag and call RASAero drag past transonic
 8
 9% find angle of attack; see photo for help
10% mathisfun.com/algebra/trig-solving-sas-triangles
11ux = u(1); 
12uy = u(2);
13umag = norm(u(1:2));
14phi = 90-r(3);
15resultant = umag.^2 + wind.^2 - 2.*umag.*wind.*cosd(phi);
16lambda = asind(wind.*sind(phi)./resultant);
17aoa = 180 - lambda - phi; %%
18
19% Calculate atmosphere
20[~, ~, rho, local_c, mu] = getAtmoConditions(r(2));
21
22% Check for mach number
23mach = umag/local_c;
24k = 1; tol = 0.01;
25while(1)
26    delta = abs(mach-aerodata(k,1));
27    if delta < tol; break; end;
28    k = k + 1;
29end
30
31% Reynolds number
32ft2m = 0.3048;
33rocket_length = 11*ft2m;
34Re_rocket_y = rho*umag*rocket_length/mu;
35
36% determine what state the rocket is in aerodynamically
37% if rocket is falling - set theta
38
39rail_length = 5.4864; % m, which is pretty ridiculous
40launch_site = 1293; % m, magic number
41aoa = 0;
42
43Sdrag = pi .* (rocket.OD.^2)./4; % axial surface area
44Slift = rocket.l.*rocket.OD;     % projected surface area
45
46if u(2) >= 0
47    
48    if r(2) <= rail_length+launch_site
49        aoa = 0;
50    end
51    
52    % Compare angle of attack and use weighted average
53    % aerodata = [RASMa, CD0, CD2, CD4, CL0, CL2, CL4, CP0, CP2, CP4];
54    if abs(aoa) <= 2
55        weight = aoa/2;
56        CD = weight * aerodata(k,3) + (1-weight) * aerodata(k,2);
57        CL = weight * aerodata(k,6) + (1-weight) * aerodata(k,5);
58        CP = weight * aerodata(k,9) + (1-weight) * aerodata(k,8);
59    elseif abs(aoa) <= 4
60        weight = (aoa-2)/2;
61        CD = weight * aerodata(k,04) + (1-weight) * aerodata(k,3);
62        CL = weight * aerodata(k,07) + (1-weight) * aerodata(k,6);
63        CP = weight * aerodata(k,10) + (1-weight) * aerodata(k,9);
64    else
65        CD = 1e10; % yikes
66        CL = 1e10; % yikes
67        CP = 1e10; % yikes
68        disp('yo, aoa > 4')
69    end
70    
71    Fdrag = CD .* Sdrag .* 0.5 .* rho .* (cosd(aoa) .* umag).^2;
72    Flift = CL .* Slift .* 0.5 .* rho .* (sind(aoa) .* umag).^2;
73    
74elseif u(2) < rocket.drogue.deploy_u && r(2) > rocket.main.deploy_h
75    CD = 0;
76    CL = 1;
77    CP = aerodata(1,8);
78    Fdrag = rocket.drogue.Cd.*rocket.drogue.S.*0.5.*rho.*uy.^2;
79    Flift = CL.*Slift.*0.5.*rho.*(ux-wind).^2; 
80    
81elseif r(2) < rocket.main.deploy_h
82    CD = 0;
83    CL = 1;
84    CP = aerodata(1,8);
85    Fdrag = rocket.main.Cd.*rocket.main.S.*0.5.*rho.*uy.^2;
86    Flift = CL.*Slift.*0.5.*rho.*(ux-wind).^2;
87else
88    CD = 0;
89    CL = 0;
90    CP = aerodata(1,8);
91    Fdrag = 0;
92    Flift = 0;
93end
94
95end
