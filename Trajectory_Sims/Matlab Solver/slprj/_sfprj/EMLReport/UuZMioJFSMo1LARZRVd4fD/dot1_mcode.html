1function [forceout, statedot] = dot(t, state, t_step, T, mass, wind, aerodata, CM)
 2
 3% avoid loading too many things. happens once each round of iterations
 4workspace = load('import2simulink.mat');
 5rocket = workspace.rocket;
 6
 7% if rocket is at apogee, set theta = -180deg
 8
 9statedot = zeros(6,1);      % initializes an empty vector
10statedot(1:3) = state(4:6); % vx, vy, omega
11
12% lookup current mass and calculate moment of inertia
13if t <= mass(end, 1)
14    k = 1; tol = t_step;
15    while(1)
16        delta = abs(mass(k,1)-t);
17        if delta < tol; break; end
18        k = k+1;
19    end
20    current_mass = mass(k,2);
21else
22    current_mass = mass(end,2);
23end
24
25I = 0.5*current_mass*rocket.OD.^2/4; % hand waves intensely
26
27% find force/acceleration; r = state(1:3); u = state(4:6)
28[f_x, f_y, moment, drag, lift, CD, CL, aoa, W] = forces(t, t_step, state(1:3), state(4:6), T,...
29    current_mass, wind, aerodata, rocket, CM);
30forceout = [f_x, f_y, moment, drag, lift, CD, CL, aoa, W];
31
32% find acceleration
33a_x = f_x/current_mass;
34a_y = f_y/current_mass;
35alpha = moment/I;
36statedot(4) = a_x;
37statedot(5) = a_y;
38statedot(6) = alpha;
39
40end
