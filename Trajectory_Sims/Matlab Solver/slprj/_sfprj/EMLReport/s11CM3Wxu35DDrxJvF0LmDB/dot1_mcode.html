1function statedot = dot(t, t_step, state, rocket, T, mass)
 2
 3% if rocket is at apogee, set theta = -180deg
 4
 5ind = t/t_step + 1; % matlab indexes at 1
 6
 7statedot = zeros(6,1);      % initializes an empty vector
 8statedot(1:3) = state(4:6); % vx, vy, omega
 9
10% lookup current mass and calculate moment of inertia
11if t <= mass(end, 1) 
12    current_mass = mass(ind,2);
13else
14    current_mass = mass(end,2);
15end
16I = 0.5*current_mass*rocket.OD.^2/4; % hand waves intensely
17
18% find force/acceleration; r = state(1:3); u = state(4:6)
19[f_x, f_y, moment] = forces(rocket, t, state(1:3), state(4:6), T,...
20    current_mass);
21
22% find acceleration
23a_x = f_x/current_mass;
24a_y = f_y/current_mass;
25alpha = moment/I;
26statedot(4) = a_x;
27statedot(5) = a_y;
28statedot(6) = alpha;
29
30end
